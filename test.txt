	#!/bin/bash
	echo "Enter num"
	read n
	if (($n%2 = = 0))
	then 
		echo "Even"
	else
		echo "Odd"
	fi
chmod +x even_odd.sh
./even_odd.sh

vi biggest.sh
	#!/bin/bash
	echo "Enter 3 nums"
	read a b c
	if (( $a>=$b && $a>=$c))
	then
		echo "$a is biggest"
	elif (( $b>=$a && $b>=$c))
	then
		echo "$b is biggest"
	else
		echo "$c is biggest"
	fi
chmod +x biggest.sh
./biggest.sh
