	fi

vi biggest.sh
	#!/bin/bash
	echo "Enter 3 numbers"
	read a b c
	if (( $a>=$b && $a>=$c))
	then
		echo "$a is biggest"
	elif (( $b>=$a && $b>=$c))
	then
		echo "$b is biggest"
	else
		echo "$c is biggest"
	fi
chmod +x biggest.sh
./biggest.sh
